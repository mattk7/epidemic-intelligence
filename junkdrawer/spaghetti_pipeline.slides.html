def plot_curves(df_graph, df_info, labels, monocolor, save, method, downsample, freq, k, percentile, start):
    """
    Visualizes multiple curves with centrality and group coloring in a time-series plot.

    Parameters:
    - df_graph (pd.DataFrame): The data to plot, indexed by time.
    - df_info (pd.DataFrame): Data containing centrality and group information for each curve.
    - labels (array-like): Array of group labels for the curves.
    - monocolor (bool): If True, all curves are colored maroon; otherwise, colored by group.
    - save (bool): Whether to save the plot as an HTML file.
    - method (callable): Method used for curve distance calculation.
    - downsample (int): Frequency for downsampling the data.
    - freq (int): Frequency of curve comparison.
    - k (int): Number of clusters/groups for analysis.
    - percentile (float): Percentile for selecting central curves.
    - start (float): Start time for measuring processing duration.

    Returns:
    - fig (plotly.graph_objs.Figure): The generated plot.
    """
    
    # Prepare the figure
    print("Preparing figure for visualization...")
    colors = ['red', 'blue', 'green', 'cyan', 'yellow', 'gray']
    cmap = dict(zip(range(len(colors)), colors))
    fig = go.Figure()
    legend_groups = set()

    alphas = {i: df_info['group'].value_counts()[i] / df_info['group'].value_counts().sum() for i in np.unique(labels)}
    max_value = max(alphas.values())
    scaled_alphas = {i: value / max_value for i, value in alphas.items()}

    for curve in df_info.T[df_graph.columns].T.sort_values('centrality', ascending=True).index:
        c_gr = df_info['group'][curve]
        c_cen = df_info['centrality'][curve]
        c_alpha = scaled_alphas[c_gr]
        
        c_color = 'maroon' if monocolor else cmap[c_gr]
        
        show_legend = c_gr not in legend_groups
        if show_legend:
            legend_groups.add(c_gr)

        fig.add_trace(go.Scatter(
            name=f'Group {c_gr}' if show_legend else "",
            x=df_graph.index,
            y=df_graph[curve],
            marker=dict(color=color_to_rgba(c_color, c_cen, c_alpha)),
            line=dict(width=(c_cen / 2) ** 2),
            mode='lines',
            showlegend=show_legend,
            legendgroup=str(c_gr)
        ))

    # Title and axis labels
    print("Finalizing figure layout...")
    fig.update_layout(
        title=f'Curve Analysis: method={method.__name__}, downsample={downsample}, freq={freq}, k={k},<br>percentile={percentile}, \
time={round(time()-start, 4)}sec',
        xaxis_title='Date',
        yaxis_title='Incidence'
    )
    
    if save:
        filename = f'../images/ensamble_method-{method.__name__}_downsample-{downsample}_freq-{freq}_k-{k}_percentile-{percentile}.html'
        print("Saving figure as HTML file...")
        fig.write_html(filename)
        
    return fig
    print("Process complete.")

def plot_cbp(df_graph, df_info, labels, monocolor, save, method, downsample, freq, k, percentile, start):
    """
    Visualizes a curve box plot (CBP) highlighting median, upper, and lower percentiles for each group.

    Parameters:
    - df_graph (pd.DataFrame): The data to plot, indexed by time.
    - df_info (pd.DataFrame): Data containing centrality and group information for each curve.
    - labels (array-like): Array of group labels for the curves.
    - monocolor (bool): If True, all groups are colored maroon; otherwise, colored by group.
    - save (bool): Whether to save the plot as an HTML file.
    - method (callable): Method used for curve distance calculation.
    - downsample (int): Frequency for downsampling the data.
    - freq (int): Frequency of curve comparison.
    - k (int): Number of clusters/groups for analysis.
    - percentile (float): Percentile for selecting central curves.
    - start (float): Start time for measuring processing duration.

    Returns:
    - fig (plotly.graph_objs.Figure): The generated plot.
    """
    
    # Prepare the figure
    print("Preparing figure for visualization...")
    colors = ['red', 'blue', 'green', 'cyan', 'yellow', 'gray']
    cmap = dict(zip(range(len(colors)), colors))
    fig = go.Figure()
    
    alphas = {i: df_info['group'].value_counts()[i] / df_info['group'].value_counts().sum() for i in np.unique(labels)}
    max_value = max(alphas.values())
    scaled_alphas = {i: value / max_value for i, value in alphas.items()}
#     print(scaled_alphas)


    for group in alphas.keys():
        dfg = df_graph.T[df_info['group'] == group].T
        median_idx = df_info[df_info['group']==group].sort_values('centrality', ascending=False).index[0]
        df_median = df_graph[median_idx]
        color = 'maroon' if monocolor else cmap[group]
        
        # Lower
        fig.add_trace(go.Scatter(
            name=f'Group {group} {percentile/2} percentile',
            x=dfg.index,
            y=dfg.T.min().values,
            marker=dict(color="#444"),
            line=dict(width=0),
            mode='lines',
            showlegend=False,
            legendgroup=str(group)  # Assign to legend group
        ))

        # Upper
        fig.add_trace(go.Scatter(
            name=f'Group {group} {100-percentile/2} percentile',
            x=dfg.index,
            y=dfg.T.max().values,
            mode='lines',
            marker=dict(color="#444"),
            line=dict(width=0),
            fillcolor=color_to_rgba(color, 1, scaled_alphas[group]/(max(alphas.keys())+1)),
            fill='tonexty',
            showlegend=False,
            legendgroup=str(group)  # Assign to legend group
        ))

        # Most Central Curve
        fig.add_trace(go.Scatter(
            name=f'Group {group}',
            x=df_median.index,
            y=df_median,
            marker=dict(color=color),
            line=dict(width=1),
            mode='lines',
            showlegend=True,  # Show legend for the central curve
            legendgroup=str(group)  # Assign to legend group
        ))
        
    fig.update_layout(
        title=f'Curve Box Plot: method={method.__name__}, downsample={downsample}, freq={freq}, k={k},<br>percentile={percentile}, \
time={round(time()-start, 4)}sec',
        xaxis_title='Date',
        yaxis_title='Incidence',
        showlegend=True
    )
    
    if save:
        filename = f'../images/curveboxplot_method-{method.__name__}_downsample-{downsample}_freq-{freq}_k-{k}_percentile-{percentile}.html'
        print("Saving figure as HTML file...")
        fig.write_html(filename)
        
    return fig
    print("Process complete.")

def pipeline_curve_analysis(df, method, graph_method, downsample, freq, k, percentile, roll, monocolor=True, save=False):
    """
    A full pipeline for curve analysis, from calculating distances to visualizing the results.

    Parameters:
    - df (pd.DataFrame): Input data containing curves to analyze.
    - method (callable): Method used for calculating distance between curves.
    - graph_method (callable): Plotting function to visualize the curves.
    - downsample (int): Frequency for downsampling the data.
    - freq (int): Frequency of curve comparison.
    - k (int): Number of clusters/groups for analysis.
    - percentile (float): Percentile for selecting central curves.
    - roll (int): Number of days to roll average for graphing.
    - monocolor (bool, optional): If True, use a single color for all groups. Defaults to True.
    - save (bool, optional): Whether to save the plot as an HTML file. Defaults to False.

    Returns:
    - fig (plotly.graph_objs.Figure): The generated plot.
    """
    
    start = time()
    
    # Step 1: Get distance between curves
    print("Step 1: Calculating distance between curves...")
    fr = distance_between_curves(df, method=method, downsample=downsample, freq=freq)
    print("Distance calculation complete.")

    # Step 2: Generate groups
    print("Step 2: Generating groups...")
    labels = generate_groups(fr, k)
    print("Group generation complete.")

    # Step 3: Get centrality scores
    print("Step 3: Calculating centrality scores...")
    centrality = get_centrality(fr, labels)
    print("Centrality score calculation complete.")

    # Step 4: Get most central curves
    print("Step 4: Identifying most central curves...")
    most_central = get_most_central(labels, centrality, percentile)
    c = list()
    for cat in most_central.keys():
        c = c + most_central[cat]

    # Generate df_info for visualization
    print("Generating df_info for visualization...")
    df_info = pd.DataFrame(normalize_column(pd.Series(centrality)))
    df_info.columns = ['centrality']
    df_info['group'] = labels

    # Prepare rolling average data for graphing
    df_roll = df.rolling(roll).median().dropna()  # One week rolling average
    df_graph = df_roll[c]  # Using the most central curves only

    # Call the plot_curves function
    fig = graph_method(df_graph, df_info, labels, monocolor, save, method, downsample, freq, k, percentile, start)
    return fig
